using System.Security.Cryptography;
using System;

public class Program
{
    public static void Main()
    {
        // AES-256 için 32 byte'lık (256 bit) bir anahtar oluşturur
        byte[] key = new byte[32];
        RandomNumberGenerator.Fill(key);
        Console.WriteLine("--- appsettings.json içine eklenecek değerler ---");
        Console.WriteLine($"\"Key\": \"{Convert.ToBase64String(key)}\",");

        // AES için 16 byte'lık (128 bit) bir IV (Vektör) oluşturur
        byte[] iv = new byte[16];
        RandomNumberGenerator.Fill(iv);
        Console.WriteLine($"\"Vektor\": \"{Convert.ToBase64String(iv)}\"");
    }
}
